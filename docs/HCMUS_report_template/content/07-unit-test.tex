\subsection{Coverage Gate Script (check-coverage.sh)}
Script \texttt{ci/check-coverage.sh} kiểm tra LINE coverage từ JaCoCo report:

\begin{lstlisting}[language=bash, caption={check-coverage.sh - Coverage Gate}]
#!/usr/bin/env bash
set -euo pipefail

module="$1"
minimum="$2"
report_path="${module}/target/site/jacoco/jacoco.xml"

# Kiem tra file ton tai
if [[ ! -f "${report_path}" ]]; then
  echo "[WARN] Coverage report missing — skipping"
  exit 0
fi

# Lay LINE counter cap report-level (dong cuoi cung)
line_counter="$(grep '<counter type="LINE"' "${report_path}" | tail -1)"
covered="$(echo "${line_counter}" | sed 's/.*covered="\([0-9]*\)".*/\1/')"
missed="$(echo "${line_counter}" | sed 's/.*missed="\([0-9]*\)".*/\1/')"

# Tinh phan tram
total=$((covered + missed))
coverage_percent="$(awk -v c="${covered}" -v t="${total}" \
    'BEGIN { printf "%.2f", (c / t) * 100 }')"

echo "Module ${module} line coverage: ${coverage_percent}%"

# So sanh voi nguong
if awk -v actual="${coverage_percent}" -v threshold="${minimum}" \
    'BEGIN { exit !(actual + 0 >= threshold + 0) }'; then
  exit 0
fi

echo "Coverage gate failed: ${coverage_percent}% < ${minimum}%"
exit 1
\end{lstlisting}

\subsection{Thiết kế Coverage Gate}
\begin{itemize}[noitemsep]
    \item \textbf{Ngưỡng}: 70\% LINE coverage (cấu hình qua biến \texttt{COVERAGE\_THRESHOLD})
    \item \textbf{Report level}: Sử dụng \texttt{tail -1} để lấy LINE counter cấp \textbf{report-level} (tổng hợp toàn module), không phải method-level
    \item \textbf{Exclude common-library}: Shared library từ upstream chỉ có 33\% coverage, không phải service cần enforce
    \item \textbf{Parse}: Dùng \texttt{grep + sed} để extract attributes từ JaCoCo XML (thay thế awk bị lỗi parsing ở phiên bản cũ)
    \item \textbf{Graceful skip}: Nếu module không có executable code hoặc report thiếu, pipeline không fail mà warning và skip
\end{itemize}
